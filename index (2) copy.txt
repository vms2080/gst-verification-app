    function updateHeaderActions() {
      const activeTab = document.querySelector('.drawer-tabs .tab.active');
      const tabId = activeTab ? activeTab.getAttribute("data-tab") : "po-details";
      const actions = headerActionsMapping[tabId] || [];
      const headerActionsContainer = document.getElementById("headerActions");
      if(actions.length > 0) {
        headerActionsContainer.innerHTML = actions.map(act =>
          `<a href="${act.action}" class="header-action-btn" title="${act.label}">
             ${act.svg}
             <span>${act.label}</span>
           </a>`
        ).join("");
      } else {
        headerActionsContainer.innerHTML = "";
      }
    }
    
    function updateFooter() {
      const activeTab = document.querySelector('.drawer-tabs .tab.active');
      if (!activeTab) return;
      const tabId = activeTab.getAttribute("data-tab");
      document.querySelectorAll('.drawer-footer').forEach(footer => {
        footer.style.display = (footer.getAttribute("data-tab-footer") === tabId) ? "block" : "none";
      });
    }
